# [boj] 1406

Linked-list를 이용한 원소의 빠른 삽입/삭제를 이용하는 문제입니다.

```cpp
#include <iostream>
#include <list>
using namespace std;
int main()
{
    int n;
    string str;
    cin >> str;
    list<char> lt;
    list<char>::iterator cursor;
    
    for(int i = 0;i<str.length();i++)
    {
        lt.push_back(str[i]);
    }
    cursor = lt.end();
    scanf("%d",&n);
    for (int t = 0;t<n;t++)
    {
        char cmd;
        cin >> cmd;
        if(cmd == 'L')
        {
            if(cursor != lt.begin())
            {
                cursor--;
            }
        }
        else if(cmd == 'D')
        {
            if(cursor != lt.end())
                cursor++;
        }
        else if(cmd == 'B')
        {
            if(cursor != lt.begin() && lt.begin() != lt.end())
                cursor = lt.erase(--cursor);
        }
        else if(cmd == 'P')
        {
            char tmp;
            cin >> tmp;
            lt.insert(cursor, tmp);
        }
    }
    list<char>::iterator iter;
    for (iter = lt.begin(); iter != lt.end(); iter++){
        cout << *iter;
    }
    return 0;
}
```
